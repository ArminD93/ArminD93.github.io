def Conv2D_block(y, number_of_layers, filters, kernel_size):

for i in range(number_of_layers):
    y = Conv2D(filters= filters, kernel_size= kernel_size, activation='relu', padding='same')(y)
    y = BatchNormalization()(y)
return y
